// api/admin/login.js — Admin authentication. // Primary mode: a single master password set in the ADMIN_PASSWORD env var // (Vercel → Settings → Environment Variables). The email field is ignored // in this mode — only the password matters. // Fallback: Supabase accounts whose profiles.role is admin/super_admin. export const config = { runtime: "edge" }; import { preflight, json, envReady, authSignIn, getProfile } from "../_lib/supabase.js"; export default async function handler(req) { if (req.method === "OPTIONS") return preflight(); if (req.method !== "POST") return json({ error: "Method not allowed" }, 405); try { const { email, password } = await req.json(); if (!password) return json({ error: "Password required" }, 400); // ── Master password mode ────────────────────────────────────────────── const master = process.env.ADMIN_PASSWORD; if (master) { if (password === master) { return json({ token: master, user: { id: "master-admin", email: email || "admin@cartchat", name: "CartChat Admin", role: "super_admin" }, }); } // Master password is configured but didn't match — try Supabase below. } // ── Supabase admin-account mode ─────────────────────────────────────── if (!envReady() || !email) return json({ error: "Invalid admin credentials" }, 401); const { ok, data } = await authSignIn(email, password); if (!ok || !data.access_token) return json({ error: "Invalid admin credentials" }, 401); const profile = await getProfile(data.user.id).catch(() => null); if (!profile || !["admin", "super_admin"].includes(profile.role)) { return json({ error: "Invalid admin credentials" }, 401); } return json({ token: data.access_token, user: { id: data.user.id, email: data.user.email, name: profile.name, role: profile.role }, }); } catch (err) { return json({ error: err.message }, 500); } }
